SimpleValidate Example

Code

      <script type="text/javascript" src="http://code.jquery.com/jquery-1.5.2.min.js">
      <script type="text/javascript" src="../src/simplevalidate.js">
      <script type="text/javascript">
        $(function(){
          new SimpleValidate($("#text-validate, #email-validate"), {
              onValidating:function(message) {
                $(this).parents("fieldset:eq(0)").find(".feedback").text(message);
              },
              onInvalid:function(message) {
                $(this).parents("fieldset:eq(0)").find(".feedback").text(message);
              },
              onValid:function() {
                $(this).parents("fieldset:eq(0)").find(".feedback").text("");
              }
          });
        });
      </script>
